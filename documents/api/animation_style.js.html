class Value{
    constructor(name, value){
        this.name = name;
        this.value = value;
    }

    transformTo(element, style){
        if(style.name = this.name){
            this.toElement()
        }
    }

    toElement(element){
        element.style[this.name] = this.toString();
    }

    toString(){
        return `${this.value}`;
    }
}
